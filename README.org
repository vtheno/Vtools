        from Match import *
        @matcher(Zero,False) # Zero is class ,False is tail recursion switch 
        def toInt(self):
            return 0
        @matcher(Succ,False)
        def toInt(self):
            # self.nat is before mcs.Succ('nat') 
            return 1 + self.nat.toInt()
        one = Succ(zero)
        two = Succ(one)
        print( toInt(zero) ,toInt(two) ) # => 0 ,2
